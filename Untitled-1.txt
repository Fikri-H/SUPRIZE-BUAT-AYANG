#include <stdio.h>

int main() {
    // Mendefinisikan variabel untuk menyimpan informasi
    char nama[] = "Nama Saya"; // A Fikri Haekal 
    char alamat[] = "Alamat Saya"; // Pati Jawa Tengah 
    char sekolah[] = "Asal Sekolah Saya"; // SMK N 2 Pati

    printf("Nama: %s\n", nama);

    printf("Asal Sekolah: %s\n", sekolah);
    printf("Alasan Masuk Polines: %s\n", alasan);

}
